CREATE TABLE Jugador(
id_jugador NUMBER(10) CONSTRAINT id_jugador_pk PRIMARY KEY,
nombre VARCHAR2(50) NOT NULL,
nickname VARCHAR2(50) NOT NULL,
sueldo double(10) NOT NULL
);

CREATE TABLE Jugador_Equipo(
id_jugador NUMBER(10) NOT NULL CONSTRAINT id_jugador_fk FOREIGN KEY(Jugador),
id_equipo NUMBER(10) NOT NULL CONSTRAINT id_equipo_fk FOREIGN KEY(Equipo),
fecha_inicio date NOT NULL,
fecha_fin date NOT NULL,
CONSTRAINT jef_pk PRIMARY KEY(id_jugador, id_equipo, fecha_inicio)
);
